# Ancient Era Government Types

# Imperial Government for major ancient empires
imperial_government = {
	create_cadet_branches = yes
	rulers_should_have_dynasty = yes
	dynasty_named_realms = yes
	council = yes
	
	vassal_contract = {
		feudal_government_vassal_contract
	}
	
	character_modifier = {
		vassal_limit = 20
		domain_limit = 5
		monthly_prestige = 1.0
		same_culture_opinion = 10
	}
	
	flag = government_is_imperial
	
	opinion = {
		same_government_opinion = 5
	}
	
	valid_holdings = {
		castle_holding
		city_holding
	}
	
	required_county_holdings = {
		castle_holding
	}
	
	primary_holding = castle_holding
	
	ai_primary_priority = {
		if = {
			limit = { culture = { has_cultural_pillar = heritage_latin } }
			add = 100
		}
		if = {
			limit = { culture = { has_cultural_pillar = heritage_chinese } }
			add = 100
		}
		if = {
			limit = { culture = { has_cultural_pillar = heritage_iranian } }
			add = 50
		}
	}
}
