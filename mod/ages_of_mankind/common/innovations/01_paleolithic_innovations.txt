# Ages of Mankind - Paleolithic Era Technologies
# Basic survival and tool-making technologies for the earliest human societies

# Stone Tools - The foundation of human technology
stone_tools = {
	group = culture_group_military
	
	is_shown = {
		culture = {
			has_innovation = stone_tools
		}
	}
	
	can_progress = {
		culture = {
			has_cultural_parameter = can_learn_stone_tools
		}
	}
	
	progress_pass_cost = {
		prestige = {
			if = {
				limit = {
					culture = {
						has_cultural_parameter = advanced_stone_working
					}
				}
				add = -10
			}
			value = 50
		}
	}
	
	character_modifier = {
		prowess = 1
		monthly_prestige = 0.05
	}
	
	culture_modifier = {
		army_damage_mult = 0.02
		development_growth_factor = 0.01
	}
}
