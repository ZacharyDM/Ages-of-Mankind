# Culture Divergence Mechanics
culture_divergence = {
	# Environmental adaptation triggers
	geographic_divergence = {
		
		# Mountains encourage isolation and unique development
		mountain_adaptation = {
			trigger = {
				capital_province = {
					terrain = mountains
				}
			}
			
			effect = {
				culture = {
					add_cultural_parameter = mountain_adapted
					add_innovation_progress = {
						innovation = shelter_construction
						progress = 10
					}
				}
			}
		}
		
		# Coastal groups develop differently
		coastal_adaptation = {
			trigger = {
				capital_province = {
					is_coastal = yes
				}
			}
			
			effect = {
				culture = {
					add_cultural_parameter = coastal_adapted
					add_innovation_progress = {
						innovation = fishing_techniques
						progress = 15
					}
				}
			}
		}
		
		# Desert survival specialization  
		desert_adaptation = {
			trigger = {
				capital_province = {
					terrain = desert
				}
			}
			
			effect = {
				culture = {
					add_cultural_parameter = desert_adapted
					add_innovation_progress = {
						innovation = water_conservation
						progress = 20
					}
				}
			}
		}
	}
}
