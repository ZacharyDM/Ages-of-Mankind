# Ages of Mankind - Era Progression Events
# Events that help drive progression through different historical eras

namespace = ages_of_mankind

# Paleolithic Discovery Events

# Discovery of Fire
ages_of_mankind.0001 = {
	type = character_event
	title = ages_of_mankind.0001.t
	desc = ages_of_mankind.0001.desc
	theme = learning
	
	trigger = {
		government = band_government
		culture = {
			NOT = { has_innovation = fire_making }
		}
		OR = {
			has_trait = hunter
			has_trait = gatherer
			learning >= 8
		}
	}
	
	weight_multiplier = {
		base = 1
		
		modifier = {
			factor = 2
			has_trait = fire_keeper
		}
		
		modifier = {
			factor = 1.5
			learning >= 12
		}
		
		# More likely in cold climates
		modifier = {
			factor = 2
			capital_province = {
				OR = {
					terrain = mountains
					terrain = taiga
					terrain = tundra
				}
			}
		}
	}
	
	immediate = {
		culture = {
			add_innovation_progress = {
				innovation = fire_making
				progress = 25
			}
		}
	}
	
	option = {
		name = ages_of_mankind.0001.a
		add_trait = fire_keeper
		add_prestige = 100
		add_stress = -20
		
		# Spread knowledge to courtiers
		every_courtier = {
			limit = { 
				OR = {
					has_trait = hunter
					has_trait = gatherer
				}
			}
			add_prestige = 25
		}
	}
	
	option = {
		name = ages_of_mankind.0001.b
		add_learning_skill = 2
		add_prestige = 50
		
		culture = {
			add_innovation_progress = {
				innovation = fire_making
				progress = 15
			}
		}
	}
}

# Stone Tool Innovation
ages_of_mankind.0002 = {
	type = character_event
	title = ages_of_mankind.0002.t
	desc = ages_of_mankind.0002.desc
	theme = stewardship
	
	trigger = {
		government = band_government

			NOT = { has_innovation = stone_tools }
		}

		}
	}

		base = 1
		
		modifier = {
			factor = 2
			has_trait = stone_shaper
		}
		
		modifier = {
			factor = 1.5
			stewardship >= 12
		}
	}
	
	immediate = {
		culture = {
			add_innovation_progress = {
				innovation = stone_tools
				progress = 30
			}
		}
	}
	
	option = {
		name = ages_of_mankind.0002.a
		add_trait = stone_shaper
		add_prestige = 75
		add_prowess_skill = 1
		
		# Better tools for the band
		every_courtier = {
			limit = { has_trait = hunter }

		}
	}
	
	option = {
		name = ages_of_mankind.0002.b

		add_prestige = 40
		
		culture = {
			add_innovation_progress = {
				innovation = stone_tools

			}
		}
	}
}

# Band Leadership Challenge
ages_of_mankind.0003 = {
	type = character_event
	title = ages_of_mankind.0003.t
	desc = ages_of_mankind.0003.desc
	theme = prowess
	
	trigger = {
		government = band_government
		is_ruler = yes
		any_courtier = {
			prowess > root.prowess
			age >= 16
		}
	}
	
	weight_multiplier = {
		base = 1
		
		modifier = {
			factor = 2
			prowess <= 8
		}
		
		modifier = {
			factor = 0.5
			has_trait = hunter
		}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				prowess > root.prowess
				age >= 16
			}
			save_scope_as = challenger
		}
	}
	
	option = {
		name = ages_of_mankind.0003.a
		# Fight for leadership
		duel = {
			skill = prowess
			target = scope:challenger
			
			50 = {
				# Victory - maintain leadership
				desc = ages_of_mankind.0003.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3
				}
				
				add_prestige = 50
				add_stress = -10
				
				scope:challenger = {
					add_opinion = {
						target = root
						modifier = respect_opinion
						opinion = 15
					}
				}
			}
			
			50 = {
				# Defeat - lose leadership
				desc = ages_of_mankind.0003.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3
				}
				
				add_prestige = -100
				add_stress = 30
				
				# Challenger becomes new leader
				scope:challenger = {
					add_prestige = 150
					add_trait = hunter # Proven fighter
				}
			}
		}
	}
	
	option = {
		name = ages_of_mankind.0003.b
		# Yield leadership peacefully
		add_prestige = -50
		add_stress = -20
		
		scope:challenger = {
			add_prestige = 100
			add_opinion = {
				target = root
				modifier = respect_opinion
				opinion = 25
			}
		}
		
		# Become an advisor
		add_trait = storyteller
	}
}
